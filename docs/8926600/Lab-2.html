countries = ['China', 'India', 'USA']
gdp = [14.34, 2.87, 22.68]

plt.show()

import matplotlib.pyplot as plt
years = [2018, 2019, 2020, 2021, 2022, 2023]
growth_rate = [5, 7, 10, 8, 12, 9]
plt.plot(years, growth_rate, marker='o', linestyle='-', color='purple')
plt.title('Cosmetics Sector Growth (2018-2023)')
plt.xlabel('Year')
plt.ylabel('Growth Rate (%)')
plt.grid(True)
plt.show()
